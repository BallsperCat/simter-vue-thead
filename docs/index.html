Component simter-vue-thead

Fork me on Github. Publish on npmjs

Define table's thead tag by structured data. Such as `columns = ["X1", "X2"]` or `columns = ["X1", {label: "X2", children: ["X21", "X22"]}]`.

Install :

// browser : (umd)
<script src="https://rawgit.com/simter/simter-vue-thead/0.2.0/dist/simter-vue-thead.min.js"></script>

// node :
import thead from 'simter-vue-thead'

Example 1 : Simple Columns

// Html :
<table id="sample">
  <thead is="st-thead" :columns="columns"></thead>
</table>

// Js :
let thead = window["simter-vue-thead"].default  // browser
import thead from 'simter-vue-thead'            // node

new Vue({
  el: "#sample",
  data: {
    columns: ["X1", "X2", "X3"]
  },
  components: {
    "st-thead": thead
  }
})

Example 2 : Group Columns

// Html :
<table id="sample">
  <thead is="st-thead" :columns="columns"></thead>
</table>

// Js :
let thead = window["simter-vue-thead"].default  // browser
import thead from 'simter-vue-thead'            // node

new Vue({
  el: "#sample",
  data: {
    columns: [
      "X1",
      {
        label: "X2",
        children: ["X21", "X22"]
      },
      {
        label: "X3",
        children: ["X31", "X32"]
      }
    ]
  },
  components: {
    "st-thead": thead
  }
})

Example 3 : Group Columns (with column width)

// Html :
<table id="sample">
  <thead is="st-thead" :columns="columns"></thead>
</table>

// Js :
let colgroup = window["simter-vue-colgroup"].default  // browser
let thead = window["simter-vue-thead"].default        // browser

import colgroup from 'simter-vue-colgroup'            // node
import thead from 'simter-vue-thead'                  // node

new Vue({
  el: "#sample",
  data: {
    columns: [
      { label: "X2", width: "50px" },
      {
        label: "X2",
        children: [
          { label: "X21", width: "50px" },
          { label: "X22", width: "50px" }
        ]
      },
      {
        label: "X3",
        children: [
          { label: "X31", width: "50px" },
          { label: "X32", width: "50px" }
        ]
      }
    ]
  },
  components: {
    "st-colgroup": colgroup,
    "st-thead": thead
  }
})
